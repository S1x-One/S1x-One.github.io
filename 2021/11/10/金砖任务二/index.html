Distributor.sol
// 定义版本号
pragma solidity ^0.4.25;

// 导入同目录下的Roles.sol文件
import “./Roles.sol”;

// 创建合约 Distributor
contract Distributor {
using Roles for Roles.Role; // 调用Roles文件下的Role函数

// 声明两个事件:添加中间商、删除中间商
event DistributorAdded(address indexed account);
event DistributorRemoved(address indexed account);

// 引入私有函数
Roles.Role private _distributors;

// 构造函数_addDistributor()
constructor (address distributor) public {
_addDistributor(distributor);
}

// 定义函数修改器onlyDistributor()
modifier onlyDistributor() {
// 将数据传到isDistributor(判断中间商)函数下判断
require(isDistributor(msg.sender), “DistributorRole: caller does not have the Distributor role”);
_;
}

// 定义isDistributor() 参数:(address account) 返回值:布尔值
function isDistributor(address account) public view returns (bool){
return _distributors.has(account);
}

// 定义addDistributor() 参数:(address account) 调用判断函数onlyDistributor
function addDistributor(address account) public onlyDistributor {
_addDistributor(account);
}jav

// 定义放弃函数renounceDistributor()
function renounceDistributor) public {
_removeDistributor(msg.sender);
}

function _addDistributor(address account) internal {
_distributors.add(account); // 调用了Roles中的add函数
// 触发Add事件
emit DistributorAdded(account);
}

function _removeDistributor(address account) internal {
_distributors.remove(account); // 调用了Roles中的remove函数
// 触发Remove函数
emit DistributorRemoved(account);
}

}

// 定义版本号
pragma solidity >=0.4.22 <0.7.0;
// 指名ABI编码格式
pragma experimental ABIEncoderV2;
// 导入sol文件
import “./FoodInfoItem.sol”;
import “./Distributor.sol”;
import “./Producer.sol”;
import “./Retailer”;

// 食品工厂合约，负责具体食品溯源信息的生成
contract Trace is Producer, Distributor, Retailer {

  // 类似python字典键值对 -> "uint256":"address"
  mapping (uint256 => address) foods; // 食品溯源id映射到具体合约的映射表
  uint[] foodlist;

// 构造函数
constructor(address producer, address distributor, address retailer)

public Producer(producer)
Distributor(distributor)
Retailer(retailer)
{
}

// <生成食品溯源信息接口> 只有Producer(农场)能调用
// name => 食品名称 | traceNumber => 溯源id | traceName => 用户名称 | quality => 当前食品质量
function newFood(string name, uint256 traceNumber, string traceName, uint8 quality)
public onlyProducer returns(address) {
require(foods[traceNumber] == address(0), “traceNumber already exist”);
FoodInfoItem food = new FoodInfoItem(name, traceName, quality msg.sender);
foods[traceNumber] = food;
foodList.push(traceNumber);
return food;
}

// <食品分销过程中增加溯源信息的接口> 只有Distributor(中间商)才能调用
// traceNumber => 溯源id | traceName => 用户名称 | quality => 当前食品质量
function addTraceInfoByDistributor(uint256 traceNumber, string traceName, uint8 quality)
public onlyDistributor return(bool) {
require(foods[traceNumber] != address(0), “traceNumber does not exist”);
